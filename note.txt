let currentText = 0;
    const texts = document.querySelectorAll('.rotating-text');

    function rotateText() {
        texts[currentText].classList.add('hide');
        currentText = (currentText + 1) % texts.length;
        texts[currentText].classList.remove('hide');
    }

    setInterval(rotateText, 2000);
